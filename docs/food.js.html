  /**
   * Crea una instancia de Food.
   *
   * @param {number} row - La fila donde se coloca la comida en la cuadrícula.
   * @param {number} column - La columna donde se coloca la comida en la cuadrícula.
   */

    super(row,column);
    this.pointsfood = 10;
  }
}
